HANDLE color=GetStdHandle(STD_OUTPUT_HANDLE);

SetConsoleTextAttribute(color, k);

// color your text in Windows console mode
// colors are 0=black 1=blue 2=green and so on to 15=white  
// colorattribute = foreground + background * 16
// to get red text on yellow use 4 + 14*16 = 228=k
// light red on yellow would be 12 + 14*16 = 236=k

SetConsoleTextAttribute ( h, FOREGROUND_RED & FOREGROUND_BLUE | BACKGROUND_BLUE  | FOREGROUND_INTENSITY | BACKGROUND_INTENSITY );
